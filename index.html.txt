<script>
        // Countdown
        let timeLeft = 5;
        const countdown = setInterval(() => {
            timeLeft--;
            document.getElementById('timer').innerText = '0' + timeLeft;
            if (timeLeft <= 0) {
                clearInterval(countdown);
                document.getElementById('countdown-screen').style.display = 'none';
                document.getElementById('quiz-box').style.display = 'block';
                showQuestion();
            }
        }, 1000);

        // Quiz Data
        const quizData = [
            { q: "What matters most in a real friendship?", opt: ["Talking every hour", "Respecting boundaries", "Expensive gifts"], correct: 1, hit: "Respect is the foundation of our bond." },
            { q: "If your friend is silent, what would you do?", opt: ["Judge them", "Give space and stay kind", "Get angry"], correct: 1, hit: "Patience protects what we have." },
            { q: "For you, friendship means:", opt: ["Comfort in silence", "Only fun times", "Control"], correct: 0, hit: "Our silence is more powerful than words." }
        ];

        let currentQ = 0;
        function showQuestion() {
            const data = quizData[currentQ];
            document.getElementById("question").innerText = data.q;
            const list = document.getElementById("options-list");
            list.innerHTML = "";
            data.opt.forEach((text, i) => {
                const btn = document.createElement("button");
                btn.className = "option-btn";
                btn.innerText = text;
                btn.onclick = () => {
                    document.getElementById("feedback").innerText = (i === data.correct) ? data.hit : "True friendship values respect.";
                    setTimeout(() => {
                        currentQ++;
                        document.getElementById("feedback").innerText = "";
                        if (currentQ < quizData.length) showQuestion();
                        else revealFinal();
                    }, 2000);
                };
                list.appendChild(btn);
            });
        }

        function revealFinal() {
            document.getElementById("quiz-box").style.display = "none";
            const bCont = document.getElementById("balloons");
            bCont.style.display = "flex";

            // Single "HAPPY BIRTHDAY" balloon set
            "HAPPY BIRTHDAY".split('').forEach((char, i) => {
                if(char === ' ') return;
                const b = document.createElement('div');
                b.className = 'balloon';
                b.innerText = char;
                b.style.setProperty('--delay', (i * 0.15) + 's');
                bCont.appendChild(b);
            });

            setTimeout(() => {
                document.getElementById("final-box").style.display = "block";
                typeMessage();
            }, 3500);
        }

        function typeMessage() {
            const msg = "Dear Zainab,\nMay your birthday mark the beginning of another year of growth, wisdom, and endless possibilities. Wishing you success and serenity always.\n\nHappy Birthday! ???";
            let k = 0;
            const target = document.getElementById("typed-text");
            function type() {
                if (k < msg.length) {
                    target.innerHTML += msg.charAt(k) === '\n' ? '<br>' : msg.charAt(k);
                    k++;
                    setTimeout(type, 45);
                }
            }
            type();
        }
    </script>
